import { Component, OnInit, Input } from '@angular/core';
import { of, Subscription, Observable } from 'rxjs';
import { delay } from 'rxjs/operators';
import { FormBuilder, FormGroup, FormControl, Validators, AbstractControl } from '@angular/forms';
import { MAT_MOMENT_DATE_FORMATS, MomentDateAdapter } from '@angular/material-moment-adapter';
import { DateAdapter, MAT_DATE_FORMATS, MAT_DATE_LOCALE } from '@angular/material/core';
import { PerfilService } from '../../../core/services/perfil.service';
import { ActivatedRoute, Router } from '@angular/router';
import { Profile } from '../../../shared/interfaces';
import { BasecidadesService } from '../../../core/services/basecidades.service';
import { NotificationsService } from 'angular2-notifications';

@Component({
  selector: 'app-cadastro',
  templateUrl: './cadastro.component.html',
  styleUrls: ['./cadastro.component.css'],
  providers: [
    { provide: MAT_DATE_LOCALE, useValue: 'pt-BR' },
    { provide: DateAdapter, useClass: MomentDateAdapter, deps: [MAT_DATE_LOCALE] },
    { provide: MAT_DATE_FORMATS, useValue: MAT_MOMENT_DATE_FORMATS },
  ],
})
export class CadastroComponent implements OnInit {

  public hide = true;
  public isLinear = true;
  public isLoading: boolean = false;
  public isLoadingEstados: boolean = false;
  public isLoadingCidades: boolean = false;
  public isSubmitted = false;
  public formRegister: FormGroup;
  public nivelSenha = ['#DD2C00', '#FF6D00', '#FFD600', '#AEEA00', '#00C853'];
  public digitosSenha = 6;
  public emailPattern = "^[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,4}$";
  public imgSrc: any = [];
  public cadastrar: Subscription;
  public perfil: Profile[];
  public dadosPerfil: any = {};
  public userData: any = {};
  public dadosUser: any = [];
  public estados: any;
  public cidades: any;

  constructor(
    private formBuilder: FormBuilder,
    private route: ActivatedRoute,
    private router: Router,
    private baseCidadesService: BasecidadesService,
    private profileService: PerfilService,
    private alertaService: NotificationsService) { }

  ngOnInit() {
    /** Validação do Formulário */
    this.formValid();
    /** Retorna a lista de Estados do Brasil */
    this.getEstados();
  }

  formValid() {
    this.formRegister = this.formBuilder.group({
      /** Valida Passo 1 */
      formArray: this.formBuilder.array([
        this.formBuilder.group({
          nome: ['', Validators.required],
          data_nasc: ['', Validators.required],
          cpf: ['', Validators.required],
          telefone: ['', Validators.required],
          genero: ['', Validators.required],
        }),
        /** Valida Passo 2 */
        this.formBuilder.group({
          endereco: ['', Validators.required],
          bairro: ['', Validators.required],
          complemento: [''],
          cidade: ['', Validators.required],
          uf: ['', Validators.required],
          cep: ['', Validators.required],
        }),
        /** Valida Passo 3 */
        this.formBuilder.group({
          email: ['', Validators.compose([
            Validators.required,
            Validators.pattern(this.emailPattern)
          ])],
          senha: ['', [Validators.required, Validators.minLength(this.digitosSenha)]]
        }),
      ])
    });
  }

  /** Retorna um FormArray com o nome 'formArray'. */
  get formArray(): AbstractControl | null { return this.formRegister.get('formArray') }

  /** Verifica campos do formulário para ir a próxima etapa */
  onNextDados() {
    this.isSubmitted = true;
    if (this.formArray.invalid) {
      return;
    }
  }

  /** Método para chamar estados do banco de dados */
  getEstados() {
    this.isLoadingEstados = true;
    this.baseCidadesService.getEstados().subscribe(data => {
      this.estados = data;
    },
      err => console.log(err),
      () => this.isLoadingEstados = false
    );
  }

  /** Método invocado após (selectionChange) do estado e popula as cidades*/
  onChangeUf(event) {
    this.isLoadingCidades = true;
    let value = event.value;
    this.baseCidadesService.getCidades(value).subscribe(data => {
      this.cidades = data;
    },
      err => console.log(err),
      () => this.isLoadingCidades = false
    );
  }

  /** Salva os dados do Cadastro */
  onSave() {
    this.isSubmitted = true;
    if (this.formArray.invalid) {
      return;
    }
    this.isLoading = true;
    this.dadosUser = this.formArray.value;
    this.userData = {
      nome: this.dadosUser[0].nome,
      cpf: this.dadosUser[0].cpf,
      data_nasc: this.dadosUser[0].data_nasc,
      telefone: this.dadosUser[0].telefone,
      genero: this.dadosUser[0].genero,
      endereco: this.dadosUser[1].endereco,
      bairro: this.dadosUser[1].bairro,
      complemento: this.dadosUser[1].complemento,
      cidade: this.dadosUser[1].cidade,
      estado: this.dadosUser[1].uf,
      cep: this.dadosUser[1].cep,
      email: this.dadosUser[2].email,
      senha: this.dadosUser[2].senha
    };

    this.profileService.crateProfile(this.userData)
      .subscribe(data => {
        //  this.dadosPerfil = data;

        if (data.status) {
          this.alertaService.success('Parabéns!', data.message);
          return of({}).pipe(delay(3000)).subscribe((data) => {
            this.isLoading = false;
            this.router.navigateByUrl('home/login');
          });
        } else {
          this.alertaService.error('Atenção!', data.message);
          return of({}).pipe(delay(3000)).subscribe(() => {
            this.isLoading = false;
          });
        }

      });
  }

  onDestroy() {
    //this.baseCidadesService.unsubscribe();
  }

}
